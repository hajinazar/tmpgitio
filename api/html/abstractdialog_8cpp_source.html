148void AbstractDialog::resumeSession()
149{
150 QString filename;
151 filename = QFileDialog::getOpenFileName(
152 this, QString("Select .state file to resume"), m_opt->filePath,
153 "*.state;;*.*", 0, QFileDialog::DontUseNativeDialog);
154
155 if (!filename.isEmpty())
156 QtConcurrent::run(this, &AbstractDialog::resumeSession_, filename);
157}

159void AbstractDialog::resumeSession_(const QString& filename)
160{
161 startProgressUpdate(tr("Resuming session..."), 0, 0);
162 m_opt->tracker()->lockForWrite();
163 m_opt->tracker()->deleteAllStructures();
164 m_opt->tracker()->unlock();
165 if (!m_opt->load(filename)) {
166 stopProgressUpdate();
167 m_opt->isStarting = false;
168 return;
169 }
170 m_opt->emitStartingSession();
171
172 stopProgressUpdate();
173
174 // Emit session started signals
175 if (m_opt->readOnly)
176 m_opt->emitReadOnlySessionStarted();
177 else
178 m_opt->emitSessionStarted();
179}

188void AbstractDialog::startProgressUpdate_(const QString& text, int min, int max)
189{
190 // progMutex->lock();
