144void AbstractEditTab::updateGUIQueueInterface()
145{
146 if (!m_isInitialized) {

148 }

150 m_queueInterfaces.contains(getCurrentQueueInterface()->getIDString()) ||

162 ui_push_queueInterfaceConfig->setEnabled(true);
163 } else {
164 ui_push_queueInterfaceConfig->setEnabled(false);
165 }
166 } else {
167 ui_push_queueInterfaceConfig->setEnabled(false);
168 }

307void AbstractEditTab::populateTemplates()
308{
309 if (!m_isInitialized) {
310 return;
311 }
312
313 QString oldTemplateName = ui_combo_templates->currentText();
314
315 int optStepIndex = getCurrentOptStep();
316 ui_combo_templates->blockSignals(true);
317 ui_combo_templates->clear();
318 ui_combo_templates->insertItems(0, getTemplateNames(optStepIndex));
319
320 // Let's see if the old template name is in the new set of templates,
321 // and if it is, set that as the current template
322 int newInd = 0;
323 for (size_t i = 0; i < ui_combo_templates->count(); ++i) {
324 if (ui_combo_templates->itemText(i) == oldTemplateName) {
325 newInd = i;
326 break;
327 }
328 }
329
330 ui_combo_templates->blockSignals(false);
331 ui_combo_templates->setCurrentIndex(newInd);
332}

334void AbstractEditTab::updateEditWidget()
335{
336 if (!m_isInitialized) {
337 return;
338 }
339 int optStepIndex = getCurrentOptStep();
340 QStringList filenames = getTemplateNames(optStepIndex);
341 int templateInd = ui_combo_templates->currentIndex();
342 QString templateName = ui_combo_templates->currentText();
343 Q_ASSERT(templateInd >= 0 && templateInd < filenames.size());
344 Q_ASSERT(templateName.compare(filenames.at(templateInd)) == 0);
345
346 if (m_opt->getNumOptSteps() != ui_list_optStep->count()) {
347 populateOptStepList();
348 }
349
350 Q_ASSERT(optStepIndex >= 0 && optStepIndex < m_opt->getNumOptSteps());
351
352 // Display appropriate entry widget. Only text entry is supported
353 // by default, reimplement this function in the derived class if
354 // list entry is needed and just call
355 // AbstractEditTab::updateEditWidget to handle entries
356 // requiring text entry.
357 ui_list_edit->setVisible(false);
358 ui_edit_edit->setVisible(true);
359
360 // Update text edit widget
361 Q_ASSERT(getTemplateNames(optStepIndex).contains(templateName));
362 QString text =
363 m_opt->getTemplate(optStepIndex, templateName.toStdString()).c_str();
364
365 ui_edit_edit->blockSignals(true);
366 ui_edit_edit->setText(text);
367 ui_edit_edit->setCurrentFont(QFont("Courier"));
368 ui_edit_edit->blockSignals(false);
369}

371void AbstractEditTab::saveCurrentTemplate()
372{
373 int optStepIndex = getCurrentOptStep();
374 QStringList filenames = getTemplateNames(optStepIndex);
375 int templateInd = ui_combo_templates->currentIndex();
376 QString templateName = ui_combo_templates->currentText();
377 Q_ASSERT(templateInd >= 0 && templateInd < filenames.size());
378 Q_ASSERT(templateName.compare(filenames.at(templateInd)) == 0);
379
380 if (m_opt->getNumOptSteps() != ui_list_optStep->count())
381 populateOptStepList();
382
383 Q_ASSERT(optStepIndex >= 0 && optStepIndex < m_opt->getNumOptSteps());
384
385 // Here we only update from the text edit widget. If any templates
386 // are using the list input, reimplement this function in the
387 // derived class and handle the lists appropriately.
388 QString text = ui_edit_edit->document()->toPlainText();
389
390 m_opt->setTemplate(optStepIndex, templateName.toStdString(),
391 text.toStdString());
392
393 bool wasBlocked = ui_edit_edit->blockSignals(true);
394 ui_edit_edit->setCurrentFont(QFont("Courier"));
395 ui_edit_edit->blockSignals(wasBlocked);
396}

406void AbstractEditTab::populateOptStepList()
407{
408 if (!m_isInitialized) {
409 return;
410 }
411 ui_list_optStep->blockSignals(true);
412
413 int currentOptStep = getCurrentOptStep();
414 const int maxSteps = m_opt->getNumOptSteps();
415
416 if (currentOptStep < 0)
417 currentOptStep = 0;
418 if (currentOptStep >= maxSteps)
419 currentOptStep = maxSteps - 1;
420
421 ui_list_optStep->clear();
422 for (int i = 1; i <= maxSteps; ++i) {
423 ui_list_optStep->addItem(tr("Optimization %1").arg(i));
424 }
425
426 ui_list_optStep->blockSignals(false);
427 ui_list_optStep->setCurrentRow(currentOptStep);
428}
