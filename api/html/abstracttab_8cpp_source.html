32void AbstractTab::initialize()

41 SLOT(writeSettings(const QString&)), Qt::DirectConnection);

43 this, SLOT(writeSettings(const QString&)),
44 Qt::BlockingQueuedConnection);
45 connect(m_dialog, SIGNAL(tabsUpdateGUI()), this, SLOT(updateGUI()));
46 connect(m_dialog, SIGNAL(tabsDisconnectGUI()), this, SLOT(disconnectGUI()));
47 connect(m_dialog, SIGNAL(tabsLockGUI()), this, SLOT(lockGUI()));
48 connect(this, SIGNAL(moleculeChanged(GlobalSearch::Structure*)), m_dialog,
49 SIGNAL(moleculeChanged(GlobalSearch::Structure*)));
50 connect(this, SIGNAL(startingBackgroundProcessing()), this,
51 SLOT(setBusyCursor()), Qt::QueuedConnection);
52 connect(this, SIGNAL(finishedBackgroundProcessing()), this,
53 SLOT(clearBusyCursor()), Qt::QueuedConnection);
54
55 m_isInitialized = true;
56 emit initialized();
57}

63void AbstractTab::setBusyCursor()
64{
65 Q_ASSERT_X(QThread::currentThread() == qApp->thread(), Q_FUNC_INFO,
66 "This function cannot be called from an background thread. "
67 "Emit AbstractTab::startingBackgroundProcessing instead.");
68 qApp->setOverrideCursor(Qt::WaitCursor);
69}
70
71void AbstractTab::clearBusyCursor()
72{
73 Q_ASSERT_X(QThread::currentThread() == qApp->thread(), Q_FUNC_INFO,
74 "This function cannot be called from an background thread. "
75 "Emit AbstractTab::finishedBackgroundProcessing instead.");
76 qApp->restoreOverrideCursor();
77}
