66bool CastepFormat::read(Structure* s, const QString& filename)
67{
68 std::ifstream ifs(filename.toStdString());
69 if (!ifs) {
70 qDebug() << "Error: CASTEP output, " << filename << ", could not "
71 << "be opened!";
72 return false;
73 }
74
75 bool coordsFound = false, energyFound = false, cellFound = false;
76
77 QList<unsigned int> atomicNums;
78 QList<Vector3> coords;
79 double energy = 0;
80 double enthalpy = 0;
81 Matrix3 cellMatrix = Matrix3::Zero();
82
83 std::string line;
84 std::vector<std::string> lineSplit;
85 while (getline(ifs, line)) {
86 // This section should contain everything we need except energy
87 if (strstr(line.c_str(), "Final Configuration")) {
88 // We will break out of this while loop when we finish with it
89 while (getline(ifs, line)) {
90 // Cell matrix.
91 if (strstr(line.c_str(), "Unit Cell")) {
92 getline(ifs, line); // Should be: ---------------...
93 getline(ifs, line); // Should be: Real Lattice(A)...
94 if (!strstr(line.c_str(), "Real Lattice(A)")) {
95 qDebug() << "Error reading the real lattice in CASTEP output!"
96 << line.c_str();
97 return false;
98 }
99 // Get the cell matrix.
100 for (unsigned short i = 0; i < 3; ++i) {
101 getline(ifs, line);
102 lineSplit = split(line, ' ');
103 // It has a size of 6 because the reciprocal lattice is here also
104 if (lineSplit.size() != 6) {
105 qDebug() << "Error reading the cell matrix in CASTEP output!"
106 << line.c_str();
107 return false;
108 }
109 cellMatrix(i, 0) = atof(lineSplit[0].c_str());
110 cellMatrix(i, 1) = atof(lineSplit[1].c_str());
111 cellMatrix(i, 2) = atof(lineSplit[2].c_str());
112 }
113
114 cellFound = true;
115 }
116 // Atomic coords
117 if (strstr(line.c_str(), "Cell Contents")) {
118 getline(ifs, line); // Should be: ------------------...
119 getline(ifs, line); // Should be:
120 getline(ifs, line); // Should be: xxxxxxxxxxxxxxxxxx...
121 getline(ifs, line); // Should be: x Element Atom ...
122 getline(ifs, line); // Should be: x Number ...
123 getline(ifs, line); // Should be: ------------------...
124
125 getline(ifs, line); // Here's where the coordinates actually start!
126 while (!strstr(line.c_str(), "xxxxxxxx")) {
127 lineSplit = split(line, ' ');
128 if (lineSplit.size() != 7) {
129 qDebug() << "Error reading atomic positions in CASTEP output!"
130 << line.c_str();
131 return false;
132 }
133 atomicNums.append(ElemInfo::getAtomicNum(lineSplit[1]));
134 coords.append(Vector3(atof(lineSplit[3].c_str()),
135 atof(lineSplit[4].c_str()),
136 atof(lineSplit[5].c_str())));
137 if (!getline(ifs, line))
138 break;
139 }
140
141 coordsFound = true;
142 }
143 // Enthalpy
144 if (strstr(line.c_str(), "Final Enthalpy")) {
145 lineSplit = split(line, ' ');
146 if (lineSplit.size() != 6) {
147 qDebug() << "Error reading final enthalpy in CASTEP output!"
148 << line.c_str();
149 return false;
150 }
151 enthalpy = atof(lineSplit[4].c_str());
152
153 // If we haven't found the energy yet, set this to be the energy
154 if (fabs(energy) < 1e-8)
155 energy = enthalpy;
156
157 energyFound = true;
158
159 // This is the last thing that we need in the inner loop. Break.
160 break;
161 }
162 }
163 }
164
165 // Energy. This may be encountered several times,
166 // so it will be updated with each encounter.
167 else if (strstr(line.c_str(), "Final energy, E")) {
168 lineSplit = split(line, ' ');
169 if (line.size() < 6) {
170 qDebug() << "Error reading final energy in CASTEP output!"
171 << line.c_str();
172 return false;
173 }
174 energy = atof(lineSplit[4].c_str());
175 energyFound = true;
176 }
177 }
178
179 if (!cellFound)
180 qDebug() << "Error: cell info was not found in CASTEP output!";
181 if (!coordsFound)
182 qDebug() << "Error: atom coords not found in CASTEP output!";
183 if (!energyFound)
184 qDebug() << "Error: energy not found in CASTEP output!";
185 if (!cellFound || !coordsFound || !energyFound)
186 return false;
187
188 // Convert coords to Cartesian
189 UnitCell uc(cellMatrix);
190 for (size_t i = 0; i < coords.size(); ++i)
191 coords[i] = uc.toCartesian(coords[i]);
192
193 s->updateAndAddToHistory(atomicNums, coords, energy, enthalpy, cellMatrix);
194 return true;
195}
196}
