20bool FileUtils::removeDir(const QString& dirName)
21{
22 bool result = true;
23 QDir dir(dirName);
24
25 if (dir.exists(dirName)) {
26 Q_FOREACH (QFileInfo info,
27 dir.entryInfoList(QDir::NoDotAndDotDot | QDir::System |
28 QDir::Hidden | QDir::AllDirs | QDir::Files,
29 QDir::DirsFirst)) {
30 if (info.isDir()) {
31 result = removeDir(info.absoluteFilePath());
32 } else {
33 result = QFile::remove(info.absoluteFilePath());
34 }
35
36 if (!result) {
37 return result;
38 }
39 }
40 result = dir.rmdir(dirName);
41 }
42
43 return result;
44}
