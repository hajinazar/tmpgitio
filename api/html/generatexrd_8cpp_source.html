33bool GenerateXrd::executeGenXrdPattern(const QStringList& args,
34 const QByteArray& input,
35 QByteArray& output)
36{
37 QString program;
38 // If the GENXRDPATTERN_EXECUTABLE environment variable is set, then
39 // use that
40 QByteArray xrdExec = qgetenv("GENXRDPATTERN_EXECUTABLE");
41 if (!xrdExec.isEmpty()) {
42 program = xrdExec;
43 } else {
44// Otherwise, search in the current directory, and then ../bin
45#ifdef _WIN32
46 QString executable = "genXrdPattern.exe";
47#else
48 QString executable = "genXrdPattern";
49#endif
50 QString path = QCoreApplication::applicationDirPath();
51 if (QFile::exists(path + "/" + executable))
52 program = path + "/" + executable;
53 else if (QFile::exists(path + "/../bin/" + executable))
54 program = path + "/../bin/" + executable;
55 else {
56 qDebug() << "Error: could not find genXrdPattern executable!";
57 return false;
58 }
59 }
60
61 QProcess p;
62 p.start(program, args);
63
64 if (!p.waitForStarted()) {
65 qDebug() << "Error: The genXrdPattern executable at" << program
66 << "failed to start.";
67 return false;
68 }
69
70 // Give it the input!
71 p.write(input.data());
72
73 // Close the write channel
74 p.closeWriteChannel();
75
76 if (!p.waitForFinished()) {
77 qDebug() << "Error: " << program << "failed to finish.";
78 output = p.readAll();
79 qDebug() << "Output is as follows:\n" << output;
80 return false;
81 }
82
83 int exitStatus = p.exitStatus();
84 output = p.readAll();
85
86 if (exitStatus == QProcess::CrashExit) {
87 qDebug() << "Error:" << program << "crashed!\n";
88 qDebug() << "Output is as follows:\n" << output;
89 return false;
90 }
91
92 if (exitStatus != QProcess::NormalExit) {
93 qDebug() << "Error:" << program << "finished abnormally with exit code "
94 << exitStatus;
95 qDebug() << "Output is as follows:\n" << output;
96 return false;
97 }
98
99 // We did it!
100 return true;
101}

103bool GenerateXrd::generateXrdPattern(const Structure& s, XrdData& results,
104 double wavelength, double peakwidth,
105 size_t numpoints, double max2theta)
106{
107 // First, write the structure in CML format
108 std::stringstream cml;
109
110 if (!CmlFormat::write(s, cml)) {
111 qDebug() << "Error in" << __FUNCTION__ << ": failed to convert structure'"
112 << s.getGeneration() << "x" << s.getIDNumber() << "' to CML"
113 << "format!";
114 return false;
115 }
116
117 // Then, convert to CIF format
118 QByteArray cif;
119 if (!OBConvert::convertFormat("cml", "cif", cml.str().c_str(), cif)) {
120 qDebug() << "Error in" << __FUNCTION__ << ": failed to convert CML"
121 << "format to CIF format with obabel";
122 return false;
123 }
124
125 // Now, execute genXrdPattern with the given inputs
126 QStringList args;
127 args << "--read-from-stdin"
128 << "--wavelength=" + QString::number(wavelength)
129 << "--peakwidth=" + QString::number(peakwidth)
130 << "--numpoints=" + QString::number(numpoints)
131 << "--max2theta=" + QString::number(max2theta);
132
133 QByteArray output;
134 if (!executeGenXrdPattern(args, cif, output)) {
135 qDebug() << "Error in" << __FUNCTION__ << ": failed to run external"
136 << "genXrdPattern program";
137 return false;
138 }
139
140 // Store the results
141 results.clear();
142 bool dataStarted = false;
143
144 QStringList lines = QString(output).split(QRegExp("[\r\n]"),
145 QString::SkipEmptyParts);
146 for (const auto& line : lines) {
147 if (!dataStarted && line.contains("# 2Theta/TOF ICalc")) {
148 dataStarted = true;
149 continue;
150 }
151
152 if (dataStarted) {
153 QStringList rowData = line.split(" ", QString::SkipEmptyParts);
154 if (rowData.size() != 2) {
155 qDebug() << "Error in" << __FUNCTION__ << ": data read from"
156 << "genXrdPattern appears to be corrupt! Data is:";
157 for (const auto& lineTmp: lines)
158 qDebug() << lineTmp;
159 return false;
160 }
161 results.push_back(
162 std::make_pair(rowData[0].toDouble(), rowData[1].toDouble()));
163 }
164 }
165
166 return true;
167}
