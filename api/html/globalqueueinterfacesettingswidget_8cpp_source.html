33void GlobalQueueInterfaceSettingsWidget::updateGUI(GlobalSearch::OptBase* opt)
34{
35 // Block signals for all widgets when we update the GUI
36 QList<bool> wasBlocked;
37 QList<QWidget*> children = this->findChildren<QWidget*>();
38 for (auto& widget : children)
39 wasBlocked.append(widget->blockSignals(true));
40
41 m_ui->edit_description->setText(opt->description);
42 m_ui->edit_host->setText(opt->host);
43 m_ui->edit_rempath->setText(opt->rempath);
44 m_ui->edit_locpath->setText(opt->filePath);
45 m_ui->edit_username->setText(opt->username);
46 m_ui->spin_port->setValue(opt->port);
47 m_ui->cb_cleanRemoteOnStop->setChecked(opt->cleanRemoteOnStop());
48 m_ui->cb_logErrorDirs->setChecked(opt->m_logErrorDirs);
49 m_ui->cb_cancelJobAfterTime->setChecked(opt->cancelJobAfterTime());
50 m_ui->spin_hoursForCancelJobAfterTime->setValue(
51 opt->hoursForCancelJobAfterTime());
52
53 m_ui->spin_hoursForCancelJobAfterTime->setEnabled(
54 m_ui->cb_cancelJobAfterTime->isChecked());
55
56 // Restore the previous states of the widgets
57 for (int i = 0; i < children.size(); ++i)
58 children[i]->blockSignals(wasBlocked[i]);
59}
