49LocalQueueInterface::~LocalQueueInterface()
50{
51 for (QHash<unsigned long, LocalQueueProcess *>::iterator

53 it_end = m_processes.end();
54 it != it_end; ++it) {

65bool LocalQueueInterface::isReadyToSearch(QString* str)
66{
67 // Is a working directory specified?
68 if (m_opt->filePath.isEmpty()) {
69 *str = tr("Local working directory is not set. Check your Queue "
70 "configuration.");
71 return false;
72 }
73
74 // Can we write to the working directory?
75 QDir workingdir(m_opt->filePath);
76 bool writable = true;
77 if (!workingdir.exists()) {
78 if (!workingdir.mkpath(m_opt->filePath)) {
79 writable = false;
80 }
81 } else {
82 // If the path exists, attempt to open a small test file for writing
83 QString filename =
84 m_opt->filePath + QString("queuetest-") + QString::number(getRandUInt());
85 QFile file(filename);
86 if (!file.open(QFile::ReadWrite)) {
87 writable = false;
88 }
89 file.remove();
90 }
91 if (!writable) {
92 *str = tr("Cannot write to working directory '%1'.\n\nPlease "
93 "change the permissions on this directory or use "
94 "a different one.")
95 .arg(m_opt->filePath);
96 return false;
97 }
98
99 *str = "";
100 return true;
101}

103bool LocalQueueInterface::writeFiles(
104 Structure* s, const QHash<QString, QString>& fileHash) const
105{
106 // Create file objects
107 QList<QFile*> files;
108 QStringList filenames = fileHash.keys();
109 for (int i = 0; i < filenames.size(); i++) {
110 files.append(new QFile(s->fileName() + "/" + filenames.at(i)));
111 }
112
113 // Check that the files can be written to
114 for (int i = 0; i < files.size(); i++) {
115 if (!files.at(i)->open(QIODevice::WriteOnly | QIODevice::Text)) {
116 m_opt->error(tr("Cannot write input file %1 (file writing failure)",
117 "1 is a file path")
118 .arg(files.at(i)->fileName()));
119 qDeleteAll(files);
120 return false;
121 }
122 }
123
124 // Set up text streams
125 QList<QTextStream*> streams;
126 for (int i = 0; i < files.size(); i++) {
127 streams.append(new QTextStream(files.at(i)));
128 }
129
130 // Write files
131 for (int i = 0; i < streams.size(); i++) {
132 *(streams.at(i)) << fileHash[filenames.at(i)];
133 }
134
135 // Close files
136 for (int i = 0; i < files.size(); i++) {
137 files.at(i)->close();
138 }
139
140 // Clean up
141 qDeleteAll(streams);
142 qDeleteAll(files);
143
144 // If there are copy files, copy those to the dir as well.
145 if (!s->copyFiles().empty()) {
146 for (const auto& copyFile : s->copyFiles()) {
147 QFile infile(copyFile.c_str());
148 QString filename = QFileInfo(infile).fileName();
149 QFile outfile(s->fileName() + "/" + filename);
150 if (!infile.copy(outfile.fileName())) {
151 m_opt->error(tr("Failed to copy file %1 to %2")
152 .arg(infile.fileName())
153 .arg(outfile.fileName()));
154 return false;
155 }
156 }
157 s->clearCopyFiles();
158 }
159
160 return true;
161}

163bool LocalQueueInterface::startJob(Structure* s)
164{
165 if (s->getJobID() != 0) {
166 m_opt->warning(tr("LocalQueueInterface::startJob: Attempting to start "
167 "job for structure %1, but a JobID is already set (%2).")
168 .arg(s->getIDString())
169 .arg(s->getJobID()));
170 return false;
171 }
172 // TODO the corresponding function in Optimizer should prepend a
173 // path for e.g. windows
174 QString command = getCurrentOptimizer(s)->localRunCommand();
175
176#ifdef WIN32
177 command = "cmd.exe /C \"" + command + "\"";
178#endif // WIN32
179
180 LocalQueueProcess* proc = new LocalQueueProcess(nullptr);
181 proc->setWorkingDirectory(s->fileName());
182 if (!getCurrentOptimizer(s)->stdinFilename().isEmpty()) {
183 proc->setStandardInputFile(s->fileName() + "/" +
184 getCurrentOptimizer(s)->stdinFilename());
185 }
186 if (!getCurrentOptimizer(s)->stdoutFilename().isEmpty()) {
187 proc->setStandardOutputFile(s->fileName() + "/" +
188 getCurrentOptimizer(s)->stdoutFilename());
189 }
190 if (!getCurrentOptimizer(s)->stderrFilename().isEmpty()) {
191 proc->setStandardErrorFile(s->fileName() + "/" +
192 getCurrentOptimizer(s)->stderrFilename());
193 }
194
195 proc->start(command);
196
197#ifdef WIN32
198 unsigned long pid = proc->pid()->dwProcessId;
199#else // WIN32
200 unsigned long pid = proc->pid();
201#endif // WIN32
202
203 s->startOptTimer();
204 s->setJobID(pid);
205
206 m_processes.insert(pid, proc);
207
208 return true;
209}

211bool LocalQueueInterface::logErrorDirectory(Structure* s) const
212{
213 QString id_s, gen_s, strdir_s;
214 id_s.sprintf("%05d", s->getIDNumber());
215 gen_s.sprintf("%05d", s->getGeneration());
216 strdir_s = gen_s + "x" + id_s;
217
218 QString command, command2;
219 QProcess proc, proc2;
220#ifdef WIN32
221 command = "mkdir " + this->m_opt->filePath + "\\errorDirs\\";
222 // Does robocopy come with all windows computers?
223 command2 = "robocopy " + s->fileName() + " " +
224 this->m_opt->filePath + "\\errorDirs\\" + strdir_s;
225#else
226 command = "mkdir -p " + this->m_opt->filePath + "/errorDirs/" + strdir_s + "/";
227 command2 =
228 "cp -r " + s->fileName() + " " + this->m_opt->filePath + "/errorDirs/" + strdir_s + "/";
229#endif
230
231 proc.start(command);
232 proc.waitForFinished();
233 proc2.start(command2);
234 proc2.waitForFinished();
235
236 return true;
237}

239bool LocalQueueInterface::stopJob(Structure* s)
240{
241 QWriteLocker wLocker(&s->lock());
242
243 unsigned long pid = static_cast<unsigned long>(s->getJobID());
244
245 if (this->m_opt->m_logErrorDirs && (s->getStatus() == Structure::Error ||
246 s->getStatus() == Structure::Restart)) {
247 logErrorDirectory(s);
248 }
249
250 if (pid == 0) {
251 // The job is not running, so just return
252 return true;
253 }
254
255 // Look-up process instance
256 LocalQueueProcess* proc = m_processes.value(pid, 0);
257 if (!proc) {
258 // No process is associated with this JobID.
259 return true;
260 }
261
262 // Kill job
263 proc->kill();
264 s->setJobID(0);
265 s->stopOptTimer();
266 m_processes.remove(pid);
267 return true;
268}

270QueueInterface::QueueStatus LocalQueueInterface::getStatus(Structure* s) const
271{
272 // lock Structure
273 QReadLocker wlocker(&s->lock());
274
275 // Look-up process instance
276 unsigned long pid = static_cast<unsigned long>(s->getJobID());
277
278 // If jobID = 0 and structure is not in "Submitted" state, return an error.
279 if (!pid && s->getStatus() != Structure::Submitted) {
280 return QueueInterface::Error;
281 }
282
283 // Look-up process instance
284 LocalQueueProcess* proc = m_processes.value(pid, 0);
285
286 // If structure is submitted, check if it's process is in the
287 // lookup table. If not, check if the completion file has been
288 // written.
289 //
290 // If the completion file exists, then the job finished before the
291 // queue checks could see it, and the function will continue on to
292 // the status checks below.
293 //
294 // Is the structure in Submitted state?
295 if (s->getStatus() == Structure::Submitted) {
296 // If the process isn't in the table
297 if (pid == 0 || proc == 0) {
298 // Is the output file exist absent?
299 bool exists;
300 getCurrentOptimizer(s)->checkIfOutputFileExists(s, &exists);
301 if (!exists) {
302 // The output file does not exist -- the job is still
303 // pending.
304 return QueueInterface::Pending;
305 }
306 }
307 // The job is either running or finished
308 return QueueInterface::Started;
309 }
310
311 // If the process is not in the table, return an error
312 if (!proc) {
313 return QueueInterface::Error;
314 }
315
316 // Note that this is not part of QProcess - status() is defined in
317 // LocalQueueProcess
318 switch (proc->status()) {
319 case LocalQueueProcess::NotStarted:
320 return QueueInterface::Pending;
321 case LocalQueueProcess::Running:
322 return QueueInterface::Running;
323 case LocalQueueProcess::Finished:
324 // Was the run successful?
325 if (proc->exitCode() != 0) {
326 m_opt->warning(tr("%1: Structure %2, PID=%3 failed. QProcess error "
327 "code: %4. Process exit code: %5 errStr: %6\n"
328 "stdout:\n%7\nstderr:\n%8")
329 .arg(Q_FUNC_INFO)
330 .arg(s->getIDString())
331 .arg(pid)
332 .arg(proc->error())
333 .arg(proc->exitCode())
334 .arg(proc->errorString())
335 .arg(QString(proc->readAllStandardOutput()))
336 .arg(QString(proc->readAllStandardError())));
337 return QueueInterface::Error;
338 }
339 bool success;
340 getCurrentOptimizer(s)->checkForSuccessfulOutput(s, &success);
341 if (success) {
342 return QueueInterface::Success;
343 } else {
344 return QueueInterface::Error;
345 }
346 default:
347 // Shouldn't reach this point...
348 return QueueInterface::Unknown;
349 }

358bool LocalQueueInterface::runACommand(const QString& workdir, const QString& command,
359 QString* sout, QString* serr, int* ercd) const
360{
361 // General note: catching the errors for local runs which rely on QProcess
362 // is tricky! E.g., the error channel might be polluted by messages from srun
363 // command which end up successful. So, we won't force a return value of False
364 // based on stderr or exit code, and only print a message if stderr is not empty
365 // or non-zero error code to help identifying possible irregularities.
366
367 QString stdout_str;
368 QString stderr_str;
369 int exitcode;
370
371 QProcess proc;
372 if (!workdir.isEmpty()) {
373 proc.setWorkingDirectory(workdir);
374 }
375 proc.start(command);
376 proc.waitForFinished(-1);
377
378 stdout_str = QString(proc.readAllStandardOutput());
379 stderr_str = QString(proc.readAllStandardError());
380 exitcode = proc.exitCode();
381
382 if (!stderr_str.isEmpty() || (exitcode != 0)) {
383 m_opt->warning(tr("Local command %1 at %2 exited with code %3 and error: %4")
384 .arg(command).arg(workdir).arg(exitcode).arg(stderr_str));
385 }
386
387 *sout = stdout_str;
388 *serr = stderr_str;
389 *ercd = exitcode;
390
391 return true;
392}

429bool LocalQueueInterface::grepFile(Structure* s, const QString& matchText,
430 const QString& filename,
431 QStringList* matches, int* exitcode,
432 const bool caseSensitive) const
433{
434 if (exitcode) {
435 *exitcode = 1;
436 }
437 Qt::CaseSensitivity ces = Qt::CaseSensitive;
438 if (!caseSensitive) {
439 ces = Qt::CaseInsensitive;
440 }
441
442 // Read the file
443 QFile infile(s->fileName() + "/" + filename);
444 if (!infile.open(QFile::ReadOnly | QFile::Text)) {
445 return false;
446 }
447
448 QTextStream in (&infile);
449 QString line;
450 do {
451 line = in.readLine();
452 if (line.contains(matchText, ces)) {
453 if (exitcode) {
454 *exitcode = 0;
455 }
456 if (matches) {
457 *matches << line;
458 }
459 }
460 } while (!line.isNull());
461
462 return true;
463}
