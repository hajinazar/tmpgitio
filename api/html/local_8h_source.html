27// Since QProcess doesn't support polling at all (e.g. the only way
28// to track its progress is to monitor signals), this class provides
29// a basic polling wrapper.
30class LocalQueueProcess : public QProcess
31{
32 Q_OBJECT
33public:
34 enum Status
35 {
36 NotStarted = 0,
37 Running,
38 Finished,
39 Error
40 };
41 LocalQueueProcess(QObject* parent) : QProcess(parent), m_status(NotStarted)
42 {
43 connect(this, SIGNAL(started()), this, SLOT(setRunning()));
44 connect(this, SIGNAL(finished(int, QProcess::ExitStatus)), this,
45 SLOT(setFinished()));
46 connect(this, SIGNAL(error(QProcess::ProcessError)), this,
47 SLOT(setFinished()));
48 }
49public slots:
50 void setRunning() { m_status = Running; };
51 void setFinished() { m_status = Finished; };
52 Status status() { return m_status; };
53private:
54 Status m_status;
55};
