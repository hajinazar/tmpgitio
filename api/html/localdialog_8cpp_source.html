37LocalQueueInterfaceConfigDialog::LocalQueueInterfaceConfigDialog(
38 AbstractDialog* parent, OptBase* opt, LocalQueueInterface* o)
39 : QDialog(parent), m_opt(opt), m_queueInterface(o), m_edit_workdir(0),
40 m_edit_description(0)
41{
42 m_vlayout = new QVBoxLayout(this);
43
44 m_label0 = new QLabel(tr("Global Queue Interface Settings"), this);
45 QFont font = m_label0->font();
46 font.setBold(true);
47 font.setPointSize(13);
48 m_label0->setFont(font);
49
50 m_top_label_layout = new QHBoxLayout();
51 m_top_label_layout->addWidget(m_label0);
52 m_vlayout->addItem(m_top_label_layout);
53
54 // Create workdir prompt
55 m_workdir_layout = new QHBoxLayout();
56
57 m_label1 = new QLabel(tr("Local working directory:"), this);
58 m_workdir_layout->addWidget(m_label1);
59
60 m_edit_workdir = new QLineEdit(this);
61 m_workdir_layout->addWidget(m_edit_workdir);
62
63 m_vlayout->addItem(m_workdir_layout);
64
65 // Create description prompt
66 m_desc_layout = new QHBoxLayout();
67
68 m_label2 = new QLabel(tr("Search description:"), this);
69 m_desc_layout->addWidget(m_label2);
70
71 m_edit_description = new QLineEdit(this);
72 m_desc_layout->addWidget(m_edit_description);
73
74 m_vlayout->addItem(m_desc_layout);
75
76 // Add checkbox
77 m_cb_logErrorDirs = new QCheckBox();
78
79 m_cb_logErrorDirs->setText("Log error directories?");
80 m_cb_logErrorDirs->setChecked(m_opt->m_logErrorDirs);
81 QString toolTip = tr("When a job fails or has to restart for any reason,"
82 "\nif this is checked, it will create a copy of the "
83 "\nfailed job directory in <localpath>/errorDirs. "
84 "\nThis can assist in figuring out why a job failed. "
85 "\nNote that if a job fails twice, the second "
86 "\nfailure will overwrite the first one.");
87 m_cb_logErrorDirs->setToolTip(toolTip);
88 m_vlayout->addWidget(m_cb_logErrorDirs);
89
90 // Add spacer
91 m_spacer =
92 new QSpacerItem(10, 10, QSizePolicy::Minimum, QSizePolicy::Expanding);
93 m_vlayout->addItem(m_spacer);
94
95 m_bbox = new QDialogButtonBox(this);
96 m_bbox->setStandardButtons(QDialogButtonBox::Ok | QDialogButtonBox::Cancel);
97 m_vlayout->addWidget(m_bbox);
98
99 setLayout(m_vlayout);
100
101 connect(m_bbox, SIGNAL(accepted()), this, SLOT(accept()));
102 connect(m_bbox, SIGNAL(rejected()), this, SLOT(reject()));
103}

105LocalQueueInterfaceConfigDialog::~LocalQueueInterfaceConfigDialog()
106{
107 // Delete all child widgets
108 QLayoutItem* child;
109 while ((child = m_vlayout->takeAt(0)) != 0) {
110 delete child->widget();
111 delete child;
112 }
113 // Delete the layout
114 delete m_vlayout;
115}
116
117void LocalQueueInterfaceConfigDialog::accept()
118{
119 m_opt->filePath = m_edit_workdir->text().trimmed();
120 m_opt->description = m_edit_description->text().trimmed();
121 m_opt->m_logErrorDirs = m_cb_logErrorDirs->isChecked();
122 QDialog::accept();
123 this->close();
124}
125
126void LocalQueueInterfaceConfigDialog::reject()
127{
128 updateGUI();
129 QDialog::reject();
130 this->close();
131}
132
133void LocalQueueInterfaceConfigDialog::updateGUI()
134{
135 m_edit_workdir->setText(m_opt->filePath);
136 m_edit_description->setText(m_opt->description);
137}
