22void Molecule::addMolecule(const Molecule& mol)
23{
24 // First, get an offset for the bond numbering later
25 size_t offset = numAtoms();
26 // Add the new atoms
27 for (const auto& atom : mol.atoms())
28 addAtom(atom);
29 // Add their bonds
30 for (const auto& bond : mol.bonds()) {
31 addBond(bond.first() + offset, bond.second() + offset, bond.bondOrder());
32 }
33}

35std::vector<Molecule> Molecule::getIndividualMolecules() const
36{
37 std::vector<Molecule> ret;
38
39 std::vector<bool> atomAlreadyUsed(numAtoms(), false);
40
41 while (true) {
42 // Find the first atom we haven't used yet
43 auto it = std::find(atomAlreadyUsed.begin(), atomAlreadyUsed.end(), false);
44
45 if (it == atomAlreadyUsed.end())
46 break;
47
48 Molecule newMol;
49 newMol.setUnitCell(unitCell());
50
51 size_t startInd = it - atomAlreadyUsed.begin();
52 newMol.addAtom(atoms()[startInd]);
53 atomAlreadyUsed[startInd] = true;
54
55 // Atoms for which to check bonds
56 std::vector<size_t> atomsToCheck(1, startInd);
57
58 // A map of the old indices to the new molecule atom indices.
59 std::unordered_map<size_t, size_t> mapToNewIndices;
60 mapToNewIndices[startInd] = 0;
61
62 // Find all atoms bonded to other atoms
63 while (!atomsToCheck.empty()) {
64 size_t checkInd = atomsToCheck[0];
65
66 for (size_t i = startInd + 1; i < numAtoms(); ++i) {
67 if (i == checkInd)
68 continue;
69
70 // bondInd will be -1 if no such bond exists
71 long long bondInd = bondBetweenAtoms(checkInd, i);
72 if (bondInd != -1 && !atomAlreadyUsed[i]) {
73 newMol.addAtom(atoms()[i]);
74 atomAlreadyUsed[i] = true;
75 atomsToCheck.push_back(i);
76 mapToNewIndices[i] = newMol.numAtoms() - 1;
77 newMol.addBond(mapToNewIndices[checkInd], mapToNewIndices[i],
78 bonds()[bondInd].bondOrder());
79 }
80 // If we have already added the atom, make sure we have added the
81 // bond
82 else if (bondInd != -1 && atomAlreadyUsed[i]) {
83 if (!newMol.areBonded(mapToNewIndices[checkInd],
84 mapToNewIndices[i])) {
85 newMol.addBond(mapToNewIndices[checkInd], mapToNewIndices[i],
86 bonds()[bondInd].bondOrder());
87 }
88 }
89 }
90 atomsToCheck.erase(atomsToCheck.begin());
91 }
92
93 ret.push_back(newMol);
94 }
95
96 return ret;
97}

99bool Molecule::removeAtom(size_t ind)
100{
101 if (ind >= m_atoms.size())
102 return false;
103 removeBondsFromAtom(ind);
104
105 // Indicate to the bonds that they should decrement any indices greater
106 // than ind.
107 for (auto& bond : m_bonds)
108 bond.atomIndexRemoved(ind);
109
110 m_atoms.erase(m_atoms.begin() + ind);
111 return true;
112}

124// We pass by copy because we want to edit a copy of newOrder...
125void Molecule::sortAtoms(std::vector<size_t> sortOrder)
126{
127 assert(sortOrder.size() == m_atoms.size());
128
129 // Only need to do m_atoms.size() - 1 since the last item will
130 // automatically be in place.
131 for (size_t i = 0; i < m_atoms.size() - 1; ++i) {
132 assert(sortOrder[i] < m_atoms.size());
133
134 // Keep swapping until the index is in the correct place
135 while (sortOrder[i] != i) {
136 size_t newInd = sortOrder[i];
137 swapAtoms(i, newInd);
138 std::swap(sortOrder[i], sortOrder[newInd]);
139 }
140 }
141}

154void Molecule::removeBondBetweenAtoms(size_t ind1, size_t ind2)
155{
156 assert(ind1 < m_atoms.size());
157 assert(ind2 < m_atoms.size());
158 for (size_t i = 0; i < m_bonds.size(); ++i) {
159 if ((m_bonds[i].first() == ind1 && m_bonds[i].second() == ind2) ||
160 (m_bonds[i].first() == ind2 && m_bonds[i].second() == ind1)) {
161 removeBond(i);
162 --i;
163 }
164 }

179{

182 for (size_t i = 0; i < m_bonds.size(); ++i) {

238void Molecule::wrapMoleculesToSmallestBonds()
239{
240 if (!hasBonds() || !hasUnitCell())
241 return;
242
243 std::vector<bool> atomAlreadyMoved(numAtoms(), false);
244
245 std::vector<size_t> atomsToCheck(1, 0);
246
247 while (!atomsToCheck.empty()) {
248 size_t checkInd = atomsToCheck[0];
249 for (size_t i = 0; i < numAtoms(); ++i) {
250 if (atomAlreadyMoved[i] || checkInd == i)
251 continue;
252
253 if (areBonded(checkInd, i)) {
254 const auto& pos1 = atom(checkInd).pos();
255 const auto& pos2 = atom(i).pos();
256 atom(i).setPos(unitCell().minimumImage(pos2 - pos1) + pos1);
257 atomAlreadyMoved[i] = true;
258 atomsToCheck.push_back(i);
259 }
260 }
261 atomsToCheck.erase(atomsToCheck.begin());
262
263 // Move on to the next group of bonded atoms if this one is done
264 if (atomsToCheck.empty()) {
265 auto it =
266 std::find(atomAlreadyMoved.begin(), atomAlreadyMoved.end(), false);
267
268 // Break if we are done
269 if (it == atomAlreadyMoved.end())
270 break;
271
272 // Otherwise, append the new atom to check and keep going
273 size_t newInd = it - atomAlreadyMoved.begin();
274 atomAlreadyMoved[newInd] = true;
275 atomsToCheck.push_back(newInd);
276 }
277 }

280double Molecule::angle(const Vector3& A, const Vector3& B, const Vector3& C)
281{
282 const Vector3& AB = A - B;
283 const Vector3& BC = C - B;
284
285 return acos(AB.dot(BC) / (AB.norm() * BC.norm())) * RAD2DEG;
286}

288double Molecule::dihedral(const Vector3& A, const Vector3& B, const Vector3& C,
289 const Vector3& D)
290{
291 const Vector3& AB = B - A;
292 const Vector3& BC = C - B;
293 const Vector3& CD = D - C;
294
295 const Vector3& n1 = AB.cross(BC);
296 const Vector3& n2 = BC.cross(CD);
297
298 return atan2(n1.cross(n2).dot(BC / BC.norm()), n1.dot(n2)) * RAD2DEG;
299}
