611inline void Molecule::addBond(size_t ind1, size_t ind2,
612 unsigned short bondOrder)
613{
614 assert(ind1 < m_atoms.size());
615 assert(ind2 < m_atoms.size());
616 // We will only allow one bond at a time between two atoms
617 if (!areBonded(ind1, ind2))
618 m_bonds.push_back(Bond(ind1, ind2, bondOrder));
