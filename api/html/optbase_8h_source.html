42//*******************************************************************
43// This part, as a whole, is to save a copy of all messages of the *
44// run to a log file, by setting up a message handler. *
45// These are here so other parts of the code (than xtalopt module) *
46// can use them. *
47// *
48// This output file will be written if XTALOPT_DEBUG flag is *
49// defined at the cmake input. *
50// *
51// The main variables/functions are: *
52// i) messageHandlerIsSet : (logical) make sure this is set once *
53// ii) gui_log_filename : name of the log file *
54// iii) customMessageOutput : the message handler function *
55// iv) saveLogFileOfRun : main function to be called *
56// *
57// The saveLogFileOfRun, whenever called, saves a copy of all the *
58// output messages to the file. This function, can and must *
59// be called only once! Either on: *
60// (1) Starting a run (in xtalopt.cpp file, startSearch()) *
61// (2) Resuming a run (in xtalopt.cpp file, load()) *
62//*******************************************************************
63static bool messageHandlerIsSet = false;
64static QString run_log_filename = "xtaloptDebug.log";
65static void customMessageOutput(QtMsgType type,
66 const QMessageLogContext &, const QString & msg)
67{
68 if (type == QtFatalMsg)
69 abort();
70
71 // Write the message to file
72 QFile outFile(run_log_filename);
73 outFile.open(QIODevice::WriteOnly | QIODevice::Append);
74 QTextStream ts(&outFile);
75 ts << msg << endl;
76 // Write the message to stdout
77 qDebug().noquote() << msg;
78}
79static void saveLogFileOfRun(QString work_dir)
80{
81 // Basically, this function is called after the filePath
82 // variable is set. Just in case, if this is not true
83 // then we will just ignore setting up the handler.
84 // Also, the logical variable messageHandler... is checked
85 // to make sure that handler is set only once in the run.
86 if (work_dir.isEmpty() || messageHandlerIsSet)
87 return;
88 // Set the log file's full path.
89 run_log_filename = work_dir + QDir::separator() + run_log_filename;
90 // Setup the message handler.
91 qInstallMessageHandler(customMessageOutput);
92 messageHandlerIsSet = true;
93}
