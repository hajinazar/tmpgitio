36Optimizer::Optimizer(OptBase* parent, const QString& filename)
37 : QObject(parent), m_opt(parent), m_hasDialog(true), m_dialog(0)
38{
39 // Set allowed data structure keys, if any, e.g.
40 // m_data.insert("Identifier name",QVariant())
41
42 // Set allowed filenames, e.g.
43 // m_templates.insert("filename.extension",QStringList)
44
45 // Setup for completion values
46 // m_completionFilename = name of file to check when opt stops
47 // m_completionStrings.clear();
48 // m_completionStrings.append("string in m_completionFilename to search for");
49 // m_completionStrings.append("Another string");
50
51 // Set output filenames to try to read data from, e.g.
52 // m_outputFilenames.append("output filename");
53 // m_outputFilenames.append("input filename");
54
55 // Set up commands
56 // Three common scenarios:
57 //
58 // VASP-esque: $ vasp
59 // Runs in working directory reading from predefined input
60 // filenames (ie. POSCAR). Set m_localRunCommand="vasp", and
61 // m_stdinFilename=m_stdoutFilename=m_stderrFilename="";
62 //
63 // GULP-esque: $ gulp < job.gin 1>job.got 2>job.err
64 //
65 // Runs in working directory using redirection to specify
66 // input/output. Set m_localRunCommand="gulp",
67 // m_stdinFilename="job.gin", m_stdoutFilename="job.got",
68 // m_stderrFilename="job.err".
69 //
70 // MOPAC-esque: $ mopac job
71 //
72 // Runs in working directory, specifying either an input filename
73 // or a base name. In both cases, put the entire command line
74 // into m_localRunCommand, ="mopac job",
75 // m_stdinFilename=m_stdoutFilename=m_stderrFilename=""
76 //
77 // m_localRunCommand = how to run this optimizer from the commandline
78 // m_stdinFilename = name of standard input file
79 // m_stdoutFilename = name of standard output file
80 // m_stderrFilename = name of standard error file
81 //
82 // Stdin/out/err is not used by default:
83 m_stdinFilename = "";
84 m_stdoutFilename = "";
85 m_stderrFilename = "";
86
87 // Set the name of the optimizer to be returned by getIDString()
88 m_idString = "Generic";
89
90 readSettings(filename);
91}

150QHash<QString, QString> Optimizer::getInterpretedTemplates(Structure* s)
151{
152 // Stop any running jobs associated with this structure
153 m_opt->queueInterface(s->getCurrentOptStep())->stopJob(s);
154
155 // Lock
156 QReadLocker locker(&s->lock());
157
158 // Check optstep info
159 int optStep = s->getCurrentOptStep();
160
161 // Unlock for optimizer calls
162 locker.unlock();
163
164 // Build hash
165 QHash<QString, QString> hash;
166 QStringList filenames =
167 m_opt->queueInterface(optStep)->getTemplateFileNames();
168 filenames.append(m_opt->optimizer(optStep)->getTemplateFileNames());
169
170 for (const auto& filename : filenames) {
171 // For debugging template issues
172 // qDebug() << "optStep is" << optStep;
173 std::string temp = m_opt->getTemplate(optStep, filename.toStdString());
174 hash.insert(filename, m_opt->interpretTemplate(temp.c_str(), s));
175 }
176
177 return hash;
178}

199bool Optimizer::checkForSuccessfulOutput(Structure* s, bool* success)
200{
201 int ec;
202 *success = false;
203 for (QStringList::const_iterator it = m_completionStrings.constBegin(),
204 it_end = m_completionStrings.constEnd();
205 it != it_end; ++it) {
206 if (!m_opt->queueInterface(s->getCurrentOptStep())
207 ->grepFile(s, (*it), m_completionFilename, 0, &ec)) {
208 qDebug() << "For structure "
209 << QString::number(s->getGeneration()) + "x" +
210 QString::number(s->getIDNumber())
211 << ":";
212 qDebug() << "The completion string, " << (*it) << ", was not found"
213 << "in the output file. Job failed.";
214 return false;
215 }
216 if (ec == 0) {
217 *success = true;
218 return true;
219 }
220 }
221 return true;
222}

224bool Optimizer::update(Structure* structure)
225{
226 // lock structure
227 QWriteLocker locker(&structure->lock());
228
229 structure->stopOptTimer();
230
231 // Copy remote files over, other prep work:
232 locker.unlock();
233 bool ok = m_opt->queueInterface(structure->getCurrentOptStep())
234 ->prepareForStructureUpdate(structure);
235 locker.relock();
236 if (!ok) {
237 m_opt->warning(
238 tr("Optimizer::update: Error while preparing to update structure %1")
239 .arg(structure->getIDString()));
240 return false;
241 }
242
243 // Try to read all files in outputFileNames
244 ok = false;
245 for (int i = 0; i < m_outputFilenames.size(); i++) {
246 if (read(structure,
247 structure->fileName() + "/" + m_outputFilenames.at(i))) {
248 ok = true;
249 break;
250 }
251 }
252 if (!ok) {
253 m_opt->warning(tr("Optimizer::Update: Error loading structure at %1")
254 .arg(structure->fileName()));
255 return false;
256 }
257
258 structure->setJobID(0);
259 locker.unlock();
260 return true;
261}

284bool Optimizer::read(Structure* structure, const QString& filename)
285{
286 // Test filename
287 QFile file(filename);
288 if (!file.open(QIODevice::ReadOnly | QIODevice::Text)) {
289 return false;
290 }
291 file.close();
292
293 if (!Formats::read(structure, filename, m_idString)) {
294 qDebug().noquote() << "Failed to read the output file " +
295 filename + " for " + structure->getIDString();
296 return false;
297 }
298
299 return true;
300}
