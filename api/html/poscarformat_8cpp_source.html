34// This is for reading a POSCAR
35bool PoscarFormat::read(Structure& s, std::istream& in)
36{
37 // First, clear the structure
38 s.clear();
39
40 size_t numLines = std::count(std::istreambuf_iterator<char>(in),
41 std::istreambuf_iterator<char>(), '\n');
42
43 // There must be at least 7 "\n"'s to have a minimum crystal
44 // (including 1 atom)
45 if (numLines < 7) {
46 std::cerr << "Error: POSCAR file is invalid: it is 7 or fewer lines\n";
47 return false;
48 }
49
50 // We have to go back to the beginning if we are going to read again
51 in.clear();
52 in.seekg(0, std::ios::beg);
53
54 // We'll use these throughout
55 string line;
56 vector<string> stringSplit;
57
58 // First line is comment line
59 getline(in, line);
60 line = trim(line);
61 string title = " ";
62 if (!line.empty())
63 title = line;
64
65 // Next line is scaling factor
66 getline(in, line);
67 double scalingFactor = atof(line.c_str());
68
69 // Next comes the matrix
70 Matrix3 cellMat;
71 for (size_t i = 0; i < 3; ++i) {
72 getline(in, line);
73 stringSplit = split(line, ' ');
74 // If this is not three, then there is some kind of error in the line
75 if (stringSplit.size() != 3) {
76 std::cerr << "Error reading lattice vectors in POSCAR\n";
77 return false;
78 }
79 // UnitCell expects a matrix of this form
80 cellMat(i, 0) = atof(stringSplit[0].c_str()) * scalingFactor;
81 cellMat(i, 1) = atof(stringSplit[1].c_str()) * scalingFactor;
82 cellMat(i, 2) = atof(stringSplit[2].c_str()) * scalingFactor;
83 }
84
85 // Sometimes, atomic symbols go here.
86 getline(in, line);
87 stringSplit = split(line, ' ');
88
89 if (stringSplit.empty()) {
90 std::cerr << "Error reading numbers of atom types in POSCAR\n";
91 return false;
92 }
93
94 // Check to see if this is an integer.
95 // If it is not, assume we have an atomic symbols list
96 bool isInt = isInteger(trim(stringSplit.at(0)));
97 vector<string> symbolsList;
98 vector<unsigned int> atomicNumbers;
99
100 if (!isInt) {
101 // Assume atomic symbols are here and store them
102 symbolsList = split(line, ' ');
103 // Store atomic nums
104 for (size_t i = 0; i < symbolsList.size(); ++i)
105 atomicNumbers.push_back(ElemInfo::getAtomicNum(symbolsList[i]));
106 // This next one should be atom types
107 getline(in, line);
108 }
109 // If the atomic symbols aren't here, try to find them in the title
110 // In Vasp 4.x, symbols are in the title like so: " O4H2 <restOfTitle>"
111 else {
112 stringSplit = split(title, ' ');
113 if (stringSplit.size() != 0) {
114 string trimFormula = trim(stringSplit.at(0));
115 // Let's replace all numbers with spaces
116 for (size_t i = 0; i < trimFormula.size(); ++i) {
117 if (isdigit(trimFormula.at(i)))
118 trimFormula[i] = ' ';
119 }
120 // Now get the symbols with a simple space split
121 symbolsList = split(trimFormula, ' ');
122 for (size_t i = 0; i < symbolsList.size(); ++i)
123 atomicNumbers.push_back(ElemInfo::getAtomicNum(symbolsList.at(i)));
124 }
125 }
126
127 stringSplit = split(line, ' ');
128 vector<unsigned int> atomCounts;
129 for (size_t i = 0; i < stringSplit.size(); ++i)
130 atomCounts.push_back(atoi(stringSplit.at(i).c_str()));
131
132 // Make sure we found the atomic numbers.
133 if (atomicNumbers.size() == 0) {
134 std::cerr << "Error: atomic numbers not found in POSCAR!\n";
135 return false;
136 }
137
138 // Make sure the numbers match
139 if (atomicNumbers.size() != atomCounts.size()) {
140 std::cerr << "Error: numSymbols and numTypes "
141 << "are not equal in POSCAR!\n";
142 return false;
143 }
144
145 // Starts with either [Ss]elective dynamics, [KkCc]artesian, or
146 // other for fractional coords.
147 getline(in, line);
148 line = trim(line);
149
150 // If selective dynamics, skip over it and get the next line
151 if (line.empty() || line.at(0) == 'S' || line.at(0) == 's')
152 getline(in, line);
153
154 line = trim(line);
155 if (line.empty()) {
156 std::cerr << "Error determining Direct or Cartesian in POSCAR\n";
157 return false;
158 }
159
160 bool cart;
161 // Check if we're using cartesian or fractional coordinates:
162 if (line[0] == 'C' || line[0] == 'c' || line[0] == 'K' || line[0] == 'k') {
163 cart = true;
164 }
165 // Assume direct if one of these was not found
166 else {
167 cart = false;
168 }
169
170 // Now get the coords
171 std::vector<Atom> atoms;
172 for (size_t i = 0; i < atomCounts.size(); ++i) {
173 for (size_t j = 0; j < atomCounts[i]; ++j) {
174 getline(in, line);
175 stringSplit = split(line, ' ');
176 // This may be greater than 3 with selective dynamics
177 if (stringSplit.size() < 3) {
178 std::cerr << "Error reading atomic coordinates in POSCAR.\n";
179 return false;
180 }
181 Vector3 coord(atof(stringSplit[0].c_str()), atof(stringSplit[1].c_str()),
182 atof(stringSplit[2].c_str()));
183 atoms.push_back(Atom(atomicNumbers[i], coord));
184 }
185 }
186
187 // Let's make a unit cell
188 UnitCell cell = UnitCell(cellMat);
189
190 // If our atomic coordinates are fractional, convert them to Cartesian
191 if (!cart) {
192 for (auto& atom : atoms)
193 atom.setPos(cell.toCartesian(atom.pos()));
194 }
195 // If they're already cartesian, we just need to apply the scaling factor
196 else {
197 for (auto& atom : atoms)
198 atom.setPos(atom.pos() * scalingFactor);
199 }
200
201 // Success! Now let's add the unit cell and the atoms!
202 s.setUnitCell(cell);
203 s.setAtoms(atoms);
204
205 return true;
206}

208bool PoscarFormat::write(const Structure& s, std::ostream& out)
209{
210 // Comment line -- set to composition then filename
211 // Construct composition
212 QStringList symbols = s.getSymbols();
213 QList<unsigned int> atomCounts = s.getNumberOfAtomsAlpha();
214 Q_ASSERT_X(symbols.size() == atomCounts.size(), Q_FUNC_INFO,
215 "getSymbols() is not the same size as getNumberOfAtomsAlpha()");
216 for (size_t i = 0; i < symbols.size(); ++i)
217 out << symbols[i].toStdString() << atomCounts[i];
218
219 out << " " << s.fileName().toStdString() << "\n";
220
221 // Scaling factor. Just 1.0
222 out << 1.0 << "\n";
223
224 // Unit Cell Vectors
225 for (uint i = 0; i < 3; i++) {
226 out << std::fixed << std::setw(12) << std::setprecision(8)
227 << s.unitCell().cellMatrix()(i, 0) << " ";
228 out << std::fixed << std::setw(12) << std::setprecision(8)
229 << s.unitCell().cellMatrix()(i, 1) << " ";
230 out << std::fixed << std::setw(12) << std::setprecision(8)
231 << s.unitCell().cellMatrix()(i, 2) << "\n";
232 }
233 // Atomic symbols
234 for (const auto& symbol : symbols)
235 out << symbol.toStdString() + " ";
236 out << "\n";
237
238 // Number of each type of atom (sorted alphabetically by symbol)
239 for (const auto& count : atomCounts)
240 out << count << " ";
241
242 out << "\n";
243 // Use fractional coordinates:
244 out << "Direct\n";
245 // Coordinates of each atom (sorted alphabetically by symbol)
246 QList<Vector3> coords = s.getAtomCoordsFrac();
247 for (const auto& coord : coords) {
248 out << std::fixed << std::setw(12) << std::setprecision(8) << coord.x()
249 << " ";
250 out << std::fixed << std::setw(12) << std::setprecision(8) << coord.y()
251 << " ";
252 out << std::fixed << std::setw(12) << std::setprecision(8) << coord.z()
253 << "\n";
254 }
255
256 return true;
257}

259void PoscarFormat::reorderAtomsToMatchPoscar(Structure& s)
260{
261 // Sort by symbols
262 const auto& symbols = s.getSymbols();
263 const auto& atoms = s.atoms();
264
265 std::vector<size_t> newOrder;
266 for (const auto& symbol_ref: symbols) {
267 for (size_t i = 0; i < atoms.size(); ++i) {
268 const auto& symbol_cur =
269 ElemInfo::getAtomicSymbol(atoms[i].atomicNumber()).c_str();
270 if (symbol_cur == symbol_ref)
271 newOrder.push_back(i);
272 }
273 }
274
275 s.reorderAtoms(newOrder);
276}
