54bool PwscfFormat::read(Structure* s, const QString& filename)
55{
56 std::ifstream ifs(filename.toStdString());
57 if (!ifs) {
58 qDebug() << "Error: PWSCF output, " << filename << ", could not "
59 << "be opened!";
60 return false;
61 }
62
63 bool coordsFound = false, energyFound = false, cellFound = false;
64
65 QList<unsigned int> atomicNums;
66 QList<Vector3> coords;
67 double energy = 0;
68 double enthalpy = 0;
69 Matrix3 cellMatrix = Matrix3::Zero();
70
71 std::string line;
72 std::vector<std::string> lineSplit;
73 while (getline(ifs, line)) {
74 // Cell parameters
75 // This section should contain the final cell and final coordinates
76 if (strstr(line.c_str(), "Begin final coordinates")) {
77 getline(ifs, line);
78 while (!strstr(line.c_str(), "End final coordinates")) {
79 if (strstr(line.c_str(), "CELL_PARAMETERS")) {
80 // Get the scaling factor and then the cell matrix
81 lineSplit = split(line, '=');
82 double scalingFactor = 1.0;
83
84 if (lineSplit.size() < 2) {
85 qDebug() << "Warning: in PWSCF output, alat was not found.";
86 qDebug() << "Assuming alat to be 1.0";
87 } else {
88 std::string alatLine = trim(lineSplit[1]);
89 replaceAll(alatLine, ")", "");
90 scalingFactor = atof(alatLine.c_str());
91 }
92
93 // Get the cell matrix
94 for (unsigned short i = 0; i < 3; ++i) {
95 getline(ifs, line);
96 lineSplit = split(line, ' ');
97 if (lineSplit.size() != 3) {
98 qDebug() << "Error reading the cell matrix in PWSCF output!"
99 << line.c_str();
100 return false;
101 }
102 cellMatrix(i, 0) = atof(lineSplit[0].c_str()) * scalingFactor;
103 cellMatrix(i, 1) = atof(lineSplit[1].c_str()) * scalingFactor;
104 cellMatrix(i, 2) = atof(lineSplit[2].c_str()) * scalingFactor;
105 }
106
107 cellFound = true;
108 }
109 // Atomic coords
110 if (strstr(line.c_str(), "ATOMIC_POSITIONS")) {
111 getline(ifs, line);
112 while (!strstr(line.c_str(), "End final coordinates")) {
113 lineSplit = split(line, ' ');
114 if (lineSplit.size() != 4) {
115 qDebug() << "Error reading atomic positions in PWSCF output!"
116 << line.c_str();
117 return false;
118 }
119 atomicNums.append(ElemInfo::getAtomicNum(lineSplit[0]));
120 coords.append(Vector3(atof(lineSplit[1].c_str()),
121 atof(lineSplit[2].c_str()),
122 atof(lineSplit[3].c_str())));
123 if (!getline(ifs, line))
124 break;
125 }
126
127 coordsFound = true;
128
129 // If we haven't found CELL_PARAMETERS, assumes that we were
130 // not relaxing the unit cell, and that our old unit cell is
131 // the same.
132 if (!cellFound) {
133 cellMatrix = s->unitCell().cellMatrix();
134 cellFound = true;
135 }
136
137 // After we find ATOMIC_POSITIONS, the loop is done. Break.
138 break;
139 }
140
141 // Get a new line. If we reached the end of the file, break.
142 if (!getline(ifs, line))
143 break;
144 }
145 }
146
147 // Enthalpy in Ry. Convert to eV.
148 else if (strstr(line.c_str(), "Final enthalpy")) {
149 lineSplit = split(line, ' ');
150 if (line.size() < 5) {
151 qDebug() << "Error reading final enthalpy in PWSCF output!"
152 << line.c_str();
153 return false;
154 }
155 enthalpy = atof(lineSplit[3].c_str()) * RY_TO_EV;
156 // If the energy hasn't been found, set the energy to be the enthalpy
157 if (fabs(energy) < 1e-8)
158 energy = enthalpy;
159 energyFound = true;
160 }
161 // Energy in Ry. Convert to eV.
162 else if (strstr(line.c_str(), "! total energy")) {
163 lineSplit = split(line, ' ');
164 if (line.size() < 6) {
165 qDebug() << "Error reading final energy in PWSCF output!"
166 << line.c_str();
167 return false;
168 }
169 energy = atof(lineSplit[4].c_str()) * RY_TO_EV;
170 energyFound = true;
171 }
172 }
173
174 if (!cellFound)
175 qDebug() << "Error: cell info was not found in PWSCF output!";
176 if (!coordsFound)
177 qDebug() << "Error: atom coords not found in PWSCF output!";
178 if (!energyFound)
179 qDebug() << "Error: energy not found in PWSCF output!";
180 if (!cellFound || !coordsFound || !energyFound)
181 return false;
182
183 // Convert coords to Cartesian
184 UnitCell uc(cellMatrix);
185 for (size_t i = 0; i < coords.size(); ++i)
186 coords[i] = uc.toCartesian(coords[i]);
187
188 s->updateAndAddToHistory(atomicNums, coords, energy, enthalpy, cellMatrix);
189 return true;
190}
191}
