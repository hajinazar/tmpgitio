57bool SSHConnectionLibSSH::isConnected()
58{
59 if (!m_session || !m_shell || !m_sftp || channel_is_closed(m_shell) ||
60 channel_is_eof(m_shell)) {
61 qWarning() << "SSHConnectionLibSSH is not connected: one or more required "
62 "channels are not initialized.";
63 return false;
64 };
65
66 if (!ssh_is_connected(m_session)) {
67 qWarning() << "SSHConnectionLibSSH is not connected.";
68 return false;
69 }
70
71 START;
72
73 // Check to see if sftp is still connected. If it is not,
74 // then reconnect it. It will sometimes disconnect...
75 if (!reconnectSftpIfNeeded()) {
76 qWarning() << "In SSHConnectionLibSSH::isConnected(),"
77 << "reconnectSftpIfNeeded() failed. Reconnecting session.";
78 return false;
79 }
80
81 END;
82 return true;

161bool SSHConnectionLibSSH::reconnectSftpIfNeeded()
162{
163 START;
164 // Check to see if the sftp is even connected
165 if (!sftpIsConnected()) {
166 bool success = reconnectSftp();
167 if (!success) {
168 qWarning() << "SSHConnectionLibSSH::reconnectSftpIfNeeded():"
169 << "reconnectSftp() failed";
170 return false;
171 }
172 // reset time stamp
173 m_sftpTimeStamp = QDateTime::currentDateTime();
174 }
175 // Check to see if 240 seconds have passed since the last reconnection
176 // if it has, reconnect the sftp
177 auto timeDifference = m_sftpTimeStamp.msecsTo(QDateTime::currentDateTime());
178
179 // reconnect the sftp every 240 seconds
180 auto interval = 240;
181 if (timeDifference >= interval * 1000) {
182#ifdef SSH_CONNECTION_LIBSSH_DEBUG
183 qDebug() << "timeDifference is " << QString::number(timeDifference);
184#endif
185 bool success = reconnectSftp();
186 if (!success) {
187 qWarning() << "SSHConnectionLibSSH::reconnectSftpIfNeeded():"
188 << "reconnectSftp() failed.";
189 return false;
190 }
191 // reset time stamp
192 m_sftpTimeStamp = QDateTime::currentDateTime();
193 }
194
195 END;
196 return true;
197}

199bool SSHConnectionLibSSH::connectSession(bool throwExceptions)
200{
201 QMutexLocker locker(&m_lock);
202 // Create session
203 m_session = ssh_new();
204 if (!m_session) {
205 if (throwExceptions) {
206 throw SSH_UNKNOWN_ERROR;
207 } else {
208 return false;
209 }
210 }
211
212 // Set options
213 int verbosity = SSH_LOG_NOLOG;
214 // int verbosity = SSH_LOG_PROTOCOL;
215 // int verbosity = SSH_LOG_PACKET;
216 int timeout = 15; // timeout in sec
217
218 ssh_options_set(m_session, SSH_OPTIONS_HOST, m_host.toStdString().c_str());
219 ssh_options_set(m_session, SSH_OPTIONS_LOG_VERBOSITY, &verbosity);
220 ssh_options_set(m_session, SSH_OPTIONS_TIMEOUT, &timeout);
221
222 if (!m_user.isEmpty()) {
223 ssh_options_set(m_session, SSH_OPTIONS_USER, m_user.toStdString().c_str());
224 }
225 ssh_options_set(m_session, SSH_OPTIONS_PORT, &m_port);
226
227 // Connect
228 if (ssh_connect(m_session) != SSH_OK) {
229 qWarning() << "SSH error: " << ssh_get_error(m_session);
230 if (throwExceptions) {
231 throw SSH_CONNECTION_ERROR;
232 } else {
233 return false;
234 }
235 }
236
237 // Verify that host is known
238 int state = ssh_is_server_known(m_session);
239 switch (state) {
240 case SSH_SERVER_KNOWN_OK:
241 break;
242 case SSH_SERVER_KNOWN_CHANGED:
243 case SSH_SERVER_FOUND_OTHER:
244 case SSH_SERVER_FILE_NOT_FOUND:
245 case SSH_SERVER_NOT_KNOWN: {
246 size_t hlen;
247 unsigned char* hash = 0;
248 ssh_key key;
249 ssh_get_publickey(m_session, &key);
250 ssh_get_publickey_hash(key, SSH_PUBLICKEY_HASH_SHA1, &hash, &hlen);
251
252 char* hexa;
253 hexa = ssh_get_hexa(hash, hlen);
254 emit unknownHostKey(QString(hexa));
255
256 // Cleanup time
257 ssh_clean_pubkey_hash(&hash);
258 ssh_key_free(key);
259
260 if (throwExceptions) {
261 throw SSH_UNKNOWN_HOST_ERROR;
262 } else {
263 return false;
264 }
265 }
266 case SSH_SERVER_ERROR:
267 qWarning() << "SSH error: " << ssh_get_error(m_session);
268 if (throwExceptions) {
269 throw SSH_UNKNOWN_ERROR;
270 } else {
271 return false;
272 }
273 }
274
275 // Authenticate
276 int rc;
277 int method;
278
279 // Try to authenticate
280 rc = ssh_userauth_none(m_session, nullptr);
281 if (rc == SSH_AUTH_ERROR) {
282 qWarning() << "SSH error: " << ssh_get_error(m_session);
283 if (throwExceptions) {
284 throw SSH_UNKNOWN_ERROR;
285 } else {
286 return false;
287 }
288 }
289
290 method = ssh_auth_list(m_session);
291 // while loop here is only so break will work. If execution gets
292 // to the end of the loop, an exception is thrown.
293 while (rc != SSH_AUTH_SUCCESS) {
294
295 // Try to authenticate with public key first
296 if (method & SSH_AUTH_METHOD_PUBLICKEY) {
297 rc = ssh_userauth_autopubkey(m_session, m_user.toStdString().c_str());
298 if (rc == SSH_AUTH_ERROR) {
299 qWarning() << "Error during auth (pubkey)";
300 qWarning() << "Error: " << ssh_get_error(m_session);
301 if (throwExceptions) {
302 throw SSH_UNKNOWN_ERROR;
303 } else {
304 return false;
305 }
306 } else if (rc == SSH_AUTH_SUCCESS) {
307 break;
308 }
309 }
310
311 // Try to authenticate with password
312 if (method & SSH_AUTH_METHOD_PASSWORD) {
313 rc = ssh_userauth_password(m_session, m_user.toStdString().c_str(),
314 m_pass.toStdString().c_str());
315 if (rc == SSH_AUTH_ERROR) {
316 qWarning() << "Error during auth (passwd)";
317 qWarning() << "Error: " << ssh_get_error(m_session);
318 if (throwExceptions) {
319 throw SSH_UNKNOWN_ERROR;
320 } else {
321 return false;
322 }
323 } else if (rc == SSH_AUTH_SUCCESS) {
324 break;
325 }
326 }
327
328 // One of the above should work, else throw an exception
329 if (throwExceptions) {
330 throw SSH_BAD_PASSWORD_ERROR;
331 } else {
332 return false;
333 }
334 }
335
336 // Open shell channel
337 if (m_shell) {
338 channel_free(m_shell);
339 m_shell = 0;
340 }
341
342 m_shell = channel_new(m_session);
343 if (!m_shell) {
344 qWarning() << "SSH error initializing shell: " << ssh_get_error(m_session);
345 if (throwExceptions) {
346 throw SSH_UNKNOWN_ERROR;
347 } else {
348 return false;
349 }
350 }
351 if (channel_open_session(m_shell) != SSH_OK) {
352 qWarning() << "SSH error opening shell: " << ssh_get_error(m_session);
353 if (throwExceptions) {
354 throw SSH_UNKNOWN_ERROR;
355 } else {
356 return false;
357 }
358 }
359 if (channel_request_shell(m_shell) != SSH_OK) {
360 qWarning() << "SSH error requesting shell: " << ssh_get_error(m_session);
361 if (throwExceptions) {
362 throw SSH_UNKNOWN_ERROR;
363 } else {
364 return false;
365 }
366 }
367
368 m_sftp = _openSFTP();
369 if (!m_sftp) {
370 qWarning() << "Could not create sftp channel.";
371 return false;
372 }
373
374 m_isValid = true;
375 END;
376 return true;
377}

387// No need to document this:
389bool SSHConnectionLibSSH::_execute(const QString& command, QString& stdout_str,
390 QString& stderr_str, int& exitcode,
391 bool printWarning)
392{
393 START;
394#ifdef SSH_CONNECTION_LIBSSH_DEBUG
395 qDebug() << "The following command is being executed:" << command;
396#endif
397 // Open new channel for exec
398 ssh_channel channel = channel_new(m_session);
399 if (!channel) {
400 if (printWarning)
401 qWarning() << "SSH error: " << ssh_get_error(m_session);
402 return false;
403 }
404 if (channel_open_session(channel) != SSH_OK) {
405 if (printWarning)
406 qWarning() << "SSH error: " << ssh_get_error(m_session);
407 channel_free(channel);
408 return false;
409 }
410
411 // Execute command
412 int ssh_exit = channel_request_exec(channel, command.toStdString().c_str());
413
414 if (ssh_exit != SSH_OK) {
415 channel_close(channel);
416 channel_free(channel);
417 return false;
418 }
419
420 // Create string streams
421 ostringstream ossout, osserr;
422
423 // Read output
424 char buffer[LIBSSH_BUFFER_SIZE];
425 int len;
426 while ((len = channel_read(channel, buffer, sizeof(buffer), 0)) > 0) {
427 ossout.write(buffer, len);
428 }
429 stdout_str = QString(ossout.str().c_str());
430 while ((len = channel_read(channel, buffer, sizeof(buffer), 1)) > 0) {
431 osserr.write(buffer, len);
432 }
433 stderr_str = QString(osserr.str().c_str());
434
435 channel_send_eof(channel);
436 channel_close(channel);
437
438 // 5 second timeout
439 int timeout = 15;
440 while (channel_get_exit_status(channel) == -1 && timeout >= 0) {
441#ifdef SSH_CONNECTION_LIBSSH_DEBUG
442 qDebug() << "Waiting for server to close channel...";
443#endif
444 GS_SLEEP(1);
445 timeout--;
446 }
447
448 exitcode = channel_get_exit_status(channel);
449
450 channel_free(channel);
451 END;
452 return true;
453}

489// No need to document this:
491bool SSHConnectionLibSSH::_copyFileToServer(const QString& localpath,
492 const QString& remotepath)
493{
494 START;
495#ifdef SSH_CONNECTION_LIBSSH_DEBUG
496 qDebug() << "copying" << localpath << "to" << remotepath;
497#endif
498
499 sftp_session sftp = m_sftp;
500 if (!sftp) {
501 qWarning() << "Could not create sftp channel.";
502 return false;
503 }
504
505 // Create input file handle
506 ifstream from(localpath.toStdString().c_str());
507 if (!from.is_open()) {
508 qWarning() << "Error opening file " << localpath << " for reading.";
509 return false;
510 }
511
512 // Create output file handle
513 sftp_file to = sftp_open(sftp, remotepath.toStdString().c_str(),
514 O_WRONLY | O_CREAT | O_TRUNC, 0750);
515 if (!to) {
516 qWarning() << "Error opening file " << remotepath << " for writing.";
517 return false;
518 }
519
520 // Create buffer
521 int size = LIBSSH_BUFFER_SIZE;
522 char* buffer = new char[size];
523
524 int readBytes;
525 while (!from.eof()) {
526 from.read(buffer, size);
527 readBytes = from.gcount();
528 if (sftp_write(to, buffer, readBytes) != readBytes) {
529 qWarning() << "Error writing to " << remotepath;
530 from.close();
531 sftp_close(to);
532 delete[] buffer;
533 return false;
534 }
535 }
536 from.close();
537 sftp_close(to);
538 delete[] buffer;
539 END;
540 return true;
541}

713 START;

780// No need to document this:
782bool SSHConnectionLibSSH::_copyDirectoryFromServer(const QString& remote,
783 const QString& local)
784{
785 START;
786#ifdef SSH_CONNECTION_LIBSSH_DEBUG
787 qDebug() << "copying" << remote << "to" << local;
788#endif
789 // Add trailing slashes:
790 QString localpath = local + "/";
791 QString remotepath = remote + "/";
792
793 sftp_dir dir;
794 sftp_attributes file;
795 sftp_session sftp = m_sftp;
796 if (!sftp) {
797 qWarning() << "Could not create sftp channel.";
798 return false;
799 }
800
801 // Open remote directory
802 dir = sftp_opendir(sftp, remotepath.toStdString().c_str());
803 if (!dir) {
804 qWarning() << "Could not open remote directory " << remotepath << ":\n\t"
805 << ssh_get_error(m_session);
806 return false;
807 }
808
809 // Create local directory
810 QDir locdir;
811 if (!locdir.mkpath(localpath)) {
812 qWarning() << "Could not create local directory " << localpath;
813 return false;
814 }
815
816 // Handle each object in the directory:
817 while ((file = sftp_readdir(sftp, dir))) {
818 if (strcmp(file->name, ".") == 0 || strcmp(file->name, "..") == 0) {
819 continue;
820 }
821
822 switch (file->type) {

1026
1027 ssh_options_set(session, SSH_OPTIONS_HOST, host.toStdString().c_str());
1028 ssh_options_set(session, SSH_OPTIONS_LOG_VERBOSITY, &verbosity);
1029 ssh_options_set(session, SSH_OPTIONS_TIMEOUT, &timeout);
1030 ssh_options_set(session, SSH_OPTIONS_PORT, &port);
1031
1032 // Connect
1033 if (ssh_connect(session) != SSH_OK) {
1034 qWarning() << "SSH error: " << ssh_get_error(session);
1035 ssh_free(session);
1036 return false;
1037 }
1038
1039 if (ssh_write_knownhost(session) < 0) {
1040 ssh_free(session);
1041 return false;
1042 }
1043
1044 ssh_free(session);
