437 virtual QString getResultsHeader(bool includeHardness, int features_num) const
438 {
439 QString out = QString("%1 %2 %3 %4")
440 .arg("Rank", 6)
441 .arg("Gen", 6)
442 .arg("ID", 6)
443 .arg("Enthalpy", 10);
444 if (includeHardness)
445 out += QString("%1")
446 .arg("Hardness", 10);
447 for (int i = 0; i< features_num; i++)
448 out += QString("%1").arg("Feature"+QString::number(i+1), 11);
449 out += QString("%1")
450 .arg("Status", 11);
451
