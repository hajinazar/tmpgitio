38// This is for reading a CONTCAR
39bool VaspFormat::read(Structure* s, const QString& filename)
40{
41 std::ifstream ifs(filename.toStdString());
42 if (!ifs) {
43 qDebug() << "Error: CONTCAR, " << filename << ", could not be opened!";
44 return false;
45 }
46
47 // First, read the POSCAR file
48 if (!PoscarFormat::read(*s, ifs)) {
49 qDebug() << "Error in VaspFormat: failed to read POSCAR file!";
50 return false;
51 }
52
53 // Now find the energy in the OUTCAR.
54 QString outcarFile = QFileInfo(filename).dir().absolutePath() + "/OUTCAR";
55
56 // Open it and make sure it exists
57 std::ifstream outcar_ifs(outcarFile.toStdString());
58
59 // We don't want to print a warning here if the OUTCAR file doesn't exist
60 // because sometimes (like when the user is loading seeds), there is no
61 // OUTCAR. We do not want the user to be concerned about it.
62
63 double energy = 0.0, enthalpy = 0.0;
64
65 bool energyFound = getOUTCAREnergy(outcar_ifs, energy);
66 bool enthalpyFound = getOUTCAREnthalpy(outcar_ifs, enthalpy);
67
68 if (outcar_ifs && !energyFound) {
69 qDebug() << "Warning: the energy could not be found in the"
70 << "OUTCAR file!";
71 }
72
73 s->setEnergy(energy);
74 if (enthalpyFound)
75 s->setEnthalpy(enthalpy);
76
77 if (s->reusePreoptBonding()) {
78 s->bonds() = s->getPreoptBonding();
79 s->clearPreoptBonding();
80 }
81
82 return true;
83}

85bool VaspFormat::getOUTCAREnergy(std::istream& in, double& energy)
86{
87 if (!in)
88 return false;
89
90 string line;
91 // We will read backwards and stop as soono as we find the energy
92 in.seekg(0, std::ios::end);
93 while (in.tellg() >= 0) {
94 reverseGetline(in, line);
95 if (strstr(line.c_str(), "free energy TOTEN")) {
96 vector<string> stringSplit = split(line, ' ');
97 // Make sure the line is long enough. If not, just keep reading.
98 if (stringSplit.size() < 5)
99 continue;
100
101 energy = atof(stringSplit[4].c_str());
102 return true;
103 } else if (strstr(line.c_str(), "free energy ML TOTEN")) { // VASP ML output
104 vector<string> stringSplit = split(line, ' ');
105 // Make sure the line is long enough. If not, just keep reading.
106 if (stringSplit.size() < 6)
107 continue;
108
109 energy = atof(stringSplit[5].c_str());
110 return true;
111 }
112 }
113 return false;
114}
