29// For writing z-matrices. Each entry contains indices of the atoms it
30// should be connected to.
31// Note: these indices are zero-based. If you write the output to a z-matrix
32// file, you should add 1 to all (except ind) because the actual z-matrix
33// format is 1-based.
34struct ZMatrixEntry
35{
36 // The index of the atom for this entry.
37 long long ind;
38
39 // The index of the atom that it is bonded to for the r index.
40 long long rInd;
41
42 // The index of the atom which is to be used with the r atom to make the
43 // angle.
44 long long angleInd;
45
46 // The index of the final atom to be used to make the dihedral.
47 long long dihedralInd;
48
49 ZMatrixEntry(long long ind = -1, long long rInd = -1, long long angleInd = -1,
50 long long dihedralInd = -1)
51 : ind(ind), rInd(rInd), angleInd(angleInd), dihedralInd(dihedralInd)
52 {
53 }
54};
